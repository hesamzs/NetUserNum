import numpy as np

def convert_to_numeric(value):
    if pd.isna(value) or value == '':
        return np.nan
    value = str(value).strip()
    if 'M' in value:
        return float(value.replace('M', '')) * 1_000_000
    elif 'k' in value:
        return float(value.replace('k', '')) * 1_000
    try:
        return float(value)
    except ValueError:
        return np.nan

    df[col] = df[col].apply(convert_to_numeric)
